#include<stdio.h>
#include<string.h>
#include<stdlib.h>
int main()
{
        char zm[26][7]={".-","-...","-.-.","-..",".","..-.","--.","....","..",".---",
        "-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--",
        "-..-","-.--","--.."};//字母的摩斯电码

        char sz[10][7]={".----","..---","...--","....-",".....","-....","--...",
        "---..","----.","-----"};//数字的摩斯电码

        char s[102];
        int i,j;
        while (gets(s)!=NULL){
        for (i=0;s[i]!='\0';i++)//转换的核心
        {
        if (s[i]>='A'&&s[i]<='Z')
        {
                printf("%s",zm[s[i]-'A']);
                if(i!=(strlen(s)-1)) printf("===");
        }
        else if (s[i]>='0'&&s[i]<='9')
        {
                printf("%s",sz[s[i]-'0']);
                if((i!=strlen(s)-1)) printf("===");
        }}printf("\n");
        }
        return 0;
}
