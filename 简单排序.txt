#include <stdio.h>

int main() {
	int arr[20];
	int s = 20;
	for (int i = 0; i < s; i++) {
		scanf("%d",&arr[i]);
	}
	int i ,t,j ;
	for (int i = 1; i < s; i++) 
		for (int j = s - 1; j >= i; j--) {
		if (arr[j] < arr[j - 1]) { 
			t = arr[j];
			arr[j] = arr[j-1];
			arr[j-1] = t; } } 
	for (int i = 0; i < s; i++) {
		printf("%d\n", arr[i]);
	}
	return 0;
}
