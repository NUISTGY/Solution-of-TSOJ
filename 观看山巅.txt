#include<iostream>
using namespace std;
int main()
{
	int n;
	long int max;
	while (cin >> n) {
		long int *arr = new long int[n];
		for (int i = 0; i < n; i++)
			cin >> arr[i];

		int num = 1;
		max = arr[0];
		for (int i = 0; i < n; i++) {
			if (arr[i] > max) {
				num++;
				max = arr[i];
			}
		}
		cout << num << endl;
		delete[] arr;
	}
	return 0;
}
