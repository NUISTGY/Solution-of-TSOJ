#include<iostream>
int gcd(int a, int b) {
	if (b == 0) {
		return a;
	}
	else {
		return gcd(b, a%b);
	}
}
int main() {
	using namespace std;
	int n;
	int array[601], brray[601];
	while (cin >> n) {
		if (n==0)
		{
			break;
		}
		for (int i = 0; i < n; i++) {
			cin >> array[i];
			brray[i] = array[i];
		}
		int sum = 0;
		for (int i = 0; i < n; i++)
			for (int j = 0; j < n; j++) {
				if ((gcd(array[i], brray[j]) == 1) && array[i] / brray[j] < 1)
					sum++;
			}

		cout << sum << endl;
	}
return 0;
}
