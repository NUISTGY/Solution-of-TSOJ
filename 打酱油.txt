#include <iostream> 
using namespace std;
 
int main(){ 
	int i, n, all=0, exam=0, exam_1=0; 
	cin>>n; 
	all = n/10; 
	for(i=0;i<=n;i+=30) { 
		exam_1 = i/30; 
		if((n-i)>50) { 
			exam_1 += 2*(n-i)/50; 
		} 
		if(exam<exam_1) {
 			exam = exam_1; 
		} 
		exam_1 = 0;
	 }
 	cout<<all+exam<<endl;
 	return 0; 
}
