#include<iostream>
int judge(int x);

int main()
{
	using namespace std;
	int n, i, j, k, count;
	while (cin>>n)
	{
		count = 0;
		for (i = 1; i < n / 2; i++)
		{
			for (j = 1 + i; j < n / 2; j++)
			{
				k = n - i - j;
				if (k > j)
					if (judge(i) && judge(k) && judge(j))
						count++;
			}
		}
		cout << count << endl;
	}
	return 0;
}
int judge(int x)
{
	int y;
	while (x)
	{
		y = x % 10;
		if (y == 2 || y == 4) return 0;
		x = x / 10;
	}

	return 1;
}
