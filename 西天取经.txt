#include <stdio.h>
int main(){
    long n,i,j,m,max,sum,book;
    while (EOF != scanf("%ld", &n)){
        sum=0;
        for ( i= 0;  i<n ; i++) {
            scanf("%ld",&j);
            max=0;
            for(m=0; j > m;m++){
                scanf("%ld",&book);
               if(max<book){
                   max=book;
               }
            }
            sum+=max;
        }
        printf("%ld\n",sum);
    }
    return 0;
}
